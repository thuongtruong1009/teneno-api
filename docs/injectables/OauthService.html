import { Injectable } from '@nestjs/common';
import { AuthProvider } from './interface/provider';

@Injectable()
export class OauthService {
    googleLogin(req: any) {
        if (!req.user) {
            return 'Not found user from google';
        }

        return {
            message: 'User information from google',
            user: req.user,
        };
    }

    githubLogin(req: any) {
        if (!req) {
            return 'Not found user from github';
        }

        return {
            message: 'User information from github',
            user: req,
        };
    }

    async findOrCreate(userId: string, provider: AuthProvider) {
        return {
            id: '42195',
            provider,
            providerId: '123',
            displayName: 'John Doe',
            photos: [
                { value: 'https://avatars.githubusercontent.com/u/28536201' },
            ],
        };
    }
}
