import { MiddlewareConsumer, Module, NestModule } from '@nestjs/common';

    AuthModule,
    AdminModule,
    UsersModule,
    FilesModule,
    PostsModule,
    ConversationsModule,
    MessagesModule,

      useClass: RolesGuard,
    },
  ],
})
